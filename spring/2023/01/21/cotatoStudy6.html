4장 화면 구성하기

요청을 보내기 위한 js파일

위와 같이 화면을 구성하는 것은 결국 사용자가 서버에 요청을 하기 위함 입니다.

사용자가 요청을 하기 위한 js파일을 작성해보겠습니다.

var main = {
    init : function (){
        var _this = this;
        $('#btn-save').on('click', function () {
            _this.save();
        });
    },
    save : function(){
        var data = {
            title: $('#title').val(),
            author: $('#author').val(),
            content: $('#content').val(),
        };
        $.ajax({
            type: 'POST',
            url: '/api/v1/posts',
            dataType: 'json',
            contentType:'application/json; charset=utf-8',
            data: JSON.stringify(data)
        }).done(function(){
            alert('글이 등록되었습니다.');
            window.location.href = '/view/';
        }).fail(function (error) {
            alert(JSON.stringify(error));
        });
    }
};
main.init();

위와 같이 js파일을 작성했습니다.
