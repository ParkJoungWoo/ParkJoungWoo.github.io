url로 데이터 읽기 심화

def get_html(url, values):
    import urllib.request
    import urllib.parse
    import chardet

    params = urllib.parse.urlencode(values)
    url_params = url + "?" + params
    data = urllib.request.urlopen(url_params).read()
    #data = 통신용 (bytes 타입)
    enc = chardet.detect(data)['encoding']
    #encoding 수행한 데이터
    return data.decode(enc)
